function my_own_pools($requestData,$count){
        global $DB, $CFG, $USER;
        $limit = " LIMIT ".$requestData['start'].",".$requestData['length'];
        if($count){
            $sql = "SELECT cat.*, qp.visible, qp.categoryid, qp.courseid, qp.poollevel
                    FROM {block_qp_instructor} AS qi
                    JOIN {block_qp} AS qp ON qp.poolid = qi.poolid
                    JOIN {question_categories} AS cat ON cat.id = qp.poolid
                    WHERE qp.poollevel = 'course' AND
                    qi.instructorid = {$USER->id}";
        }else{
            $sql = "SELECT cat.*, qp.visible, qp.categoryid, qp.courseid, qp.poollevel
                    FROM {block_qp_instructor} AS qi
                    JOIN {block_qp} AS qp ON qp.poolid = qi.poolid
                    JOIN {question_categories} AS cat ON cat.id = qp.poolid
                    WHERE qp.poollevel = 'course' AND
                    qi.instructorid = {$USER->id}
                    ORDER BY cat.id ASC".$limit;
        }
        
        $pools = $DB->get_records_sql($sql);
        return $pools;
    }
